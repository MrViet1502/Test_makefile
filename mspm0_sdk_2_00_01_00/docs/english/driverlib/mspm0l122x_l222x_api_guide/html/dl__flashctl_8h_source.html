969 __STATIC_INLINE uint32_t DL_FlashCTL_getFlashSectorNumberInBank(
970  FLASHCTL_Regs *flashctl, uint32_t addr)
971 {
972  uint8_t numBanks = DL_FactoryRegion_getNumBanks();
973  uint32_t mainFlashSize = DL_FactoryRegion_getMAINFlashSize();
974  uint32_t sector, bankSectors, sectorInBank;
975 
976  /* Current sector over the whole memory map */
977  sector = DL_FlashCTL_getFlashSectorNumber(flashctl, addr);
978  if (numBanks > 1) {
979  /* Number of sectors per bank, considered max sector
980  * count for a bank-adjusted sector. Assume banks are
981  * evenly distributed */
982  bankSectors = (mainFlashSize / numBanks);
983  /* We will not assume that the maximum number of bank sectors is a
984  * multiple of 2. Which does mean incurring a modulo operation. */
985  sectorInBank = sector % bankSectors;
986  } else {
987  sectorInBank = sector;
988  }
989 
990  return sectorInBank;
991 }
